62 #endif

64  utilities::step_size(uv_data.vis, measurements,
65  std::make_shared<sopt::LinearTransform<Vector<t_complex>> const>(Psi),
66  sara.size()) *
67  1e-3;

69  PURIFY_MEDIUM_LOG("Gamma {}", gamma);
70 
71  // shared pointer because the convergence function need access to some data that we would rather
72  // not reproduce. E.g. padmm definition is self-referential.
73  auto padmm = std::make_shared<sopt::algorithm::ImagingProximalADMM<t_complex>>(uv_data.vis);
74  padmm->itermax(50)

76  .relative_variation(1e-3)

83  .l1_proximal_adjoint_space_comm(comm)

109 #endif

125  auto convergence_function = [](const Vector<t_complex> &x) { return true; };
126  const std::shared_ptr<t_uint> iter = std::make_shared<t_uint>(0);
127  const auto algo_update = [uv_data, imsizex, imsizey, padmm_weak, iter,
128  comm](const Vector<t_complex> &x) -> bool {
129  auto padmm = padmm_weak.lock();
130  if (comm.is_root()) PURIFY_MEDIUM_LOG("Step size γ {}", padmm->gamma());
131  *iter = *iter + 1;
132  Vector<t_complex> const alpha = padmm->Psi().adjoint() * x;
133  const t_real new_gamma = comm.all_reduce(alpha.real().cwiseAbs().maxCoeff(), MPI_MAX) * 1e-3;
134  if (comm.is_root()) PURIFY_MEDIUM_LOG("Step size γ update {}", new_gamma);
135  padmm->gamma(((std::abs(padmm->gamma() - new_gamma) > 0.2) and *iter < 200) ? new_gamma
136  : padmm->gamma());
137  // updating parameter
138 
139  Vector<t_complex> const residual = padmm->Phi().adjoint() * (uv_data.vis - padmm->Phi() * x);
140 
141  if (comm.is_root()) {
142  pfitsio::write2d(x, imsizey, imsizex, "mpi_solution_update.fits");
143  pfitsio::write2d(residual, imsizey, imsizex, "mpi_residual_update.fits");
144  }
145  return true;
146  };
147  auto lambda = [convergence_function, algo_update](Vector<t_complex> const &x) {
148  return convergence_function(x) and algo_update(x);
149  };
150  padmm->is_converged(lambda);
151  return padmm;
152 }
