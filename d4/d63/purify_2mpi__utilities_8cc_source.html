25  }

28  t_int j = 0;
29  for (auto const end : ends) {

31  ++j;
32  }
33  return j;
34  };

40  ++i;
41  continue;
42  }

44  if (groups[swapper] == expected(swapper)) {
45  ++swapper;
46  continue;
47  }
48  std::swap(groups[i], groups[swapper]);
49  std::swap(uv_params.u(i), uv_params.u(swapper));
