173 std::vector<Image<t_complex>> read3d(const std::string &fits_name) {
174  /*
175  Reads in a cube from a fits file and returns the vector of images.
176  fits_name:: name of fits file
177  */
178 
179  fitsfile *fptr;
180  int status = 0;
181  PURIFY_LOW_LOG("Reading fits file {}", fits_name);
182  if (fits_open_image(&fptr, fits_name.c_str(), READONLY, &status))
183  throw std::runtime_error("Error opening image " + fits_name);
184  const t_int naxis = read_key<int>(fptr, "NAXIS", &status);
185  if (naxis < 1) throw std::runtime_error("Image contains zero axes.");
186  const t_int rows = read_key<int>(fptr, "NAXIS1", &status);
187  const t_int cols = (naxis > 1) ? read_key<int>(fptr, "NAXIS2", &status) : 1;
188  const t_int channels = (naxis > 2) ? read_key<int>(fptr, "NAXIS3", &status) : 1;
189  const t_int pols = (naxis > 3) ? read_key<int>(fptr, "NAXIS4", &status) : 1;
190  PURIFY_LOW_LOG("Axes {}", naxis);
191  std::vector<long> fpixel(naxis, 1);
192  PURIFY_LOW_LOG("Dimensions {}x{}x{}x{}", rows, cols, channels, pols);
193  if (pols > 1) throw std::runtime_error("Too many polarisations when reading " + fits_name);
194  std::vector<Image<t_complex>> eigen_images;
195  t_real nulval = 0;
196  t_int anynul = 0;
197  Vector<t_real> image = Vector<t_real>::Zero(rows * cols * channels * pols);
198  if (fits_read_pix(fptr, TDOUBLE, fpixel.data(), static_cast<long>(image.size()), &nulval,
199  image.data(), &anynul, &status))
200  throw std::runtime_error("Error reading data from image " + fits_name);
201  if (anynul) PURIFY_LOW_LOG("There are bad values when reading " + fits_name);
202  for (int i = 0; i < channels; i++) {
203  Vector<t_complex> eigen_image = Vector<t_complex>::Zero(rows * cols);
204  eigen_image.real() = image.segment(i * rows * cols, rows * cols);
205  eigen_images.push_back(Image<t_complex>::Map(eigen_image.data(), rows, cols));
206  }
207  if (fits_close_file(fptr, &status))
208  throw std::runtime_error("Problem closing fits file: " + fits_name);
209  return eigen_images;
210 }
