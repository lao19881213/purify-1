21 utilities::vis_params read_measurements(const std::vector<std::string> &names, const bool w_term,
22  const stokes pol, const utilities::vis_units units) {
23  std::vector<std::string> found_files;
24  format format_type = format::uvfits;
25  for (t_int i = 0; i < names.size(); i++) {
26  const boost::filesystem::path file_path(names.at(i));
27  if (not file_exists(file_path.native())) {
28  PURIFY_HIGH_LOG("Missing file will be removed from list: {}", names.at(i));
29  } else {
30  found_files.emplace_back(names.at(i));
31  format format_check = format::uvfits;
32  // checking if reading measurement set or .vis file
33  std::size_t found = names.at(i).find_last_of(".");
34  std::string format = "." + names.at(i).substr(found + 1);
35  std::transform(format.begin(), format.end(), format.begin(), ::tolower);
36  if (format == ".ms") {
37  format_check = format::ms;
38  if (not dir_exists(file_path.native()))
39  throw std::runtime_error(names.at(i) +
40  " is not a directory, as expected for a measurement set.");
41  } else if (format == ".vis") {
42  format_check = format::vis;
43  if (not file_exists(file_path.native()))
44  throw std::runtime_error(names.at(i) + " is not a regular file.");
45  } else if (format == ".uvfits") {
46  format_check = format::uvfits;
47  if (not file_exists(file_path.native()))
48  throw std::runtime_error(names.at(i) + " is not a regular file.");
49  } else
50  throw std::runtime_error("File extention for " + names.at(i) +
51  " not recognised. Must be .vis, .uvfits, or .ms.");
52  if (i == 0) format_type = format_check;
53  if (i > 0 and (format_check != format_type))
54  throw std::runtime_error("File extention is not the same for " + names.at(i) + " " +
55  names.at(i - 1));
56  }
57  }
58 
59  if (found_files.size() == 0) throw std::runtime_error("No files found, all files are missing!");
60  switch (format_type) {
61  case (format::vis): {
62  if (pol != stokes::I)
63  throw std::runtime_error(
64  ".vis files are ascii, so it is assumed that it is Stokes I. But, you are trying to "
65  "choose a different type!");
66  auto measurements = utilities::read_visibility(found_files, w_term);
67  measurements.units = units;
68  return measurements;
69  break;
70  }
71  case (format::uvfits): {
72  return pfitsio::read_uvfits(found_files, true, pol);
73  break;
74  }
75  case (format::ms): {
76 #ifdef PURIFY_CASACORE
77  return casa::read_measurementset(found_files, pol);
78  break;
79 #else
80  throw std::runtime_error(
81  "You want to read a measurement set, but you did not compile with casacore.");
82 #endif
83  }
84  default:
85  throw std::runtime_error("Format not recongised.");
86  }
87 }
