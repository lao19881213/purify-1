396 t_real calculate_l2_radius(const t_uint y_size, const t_real &sigma, const t_real &n_sigma,
397  const std::string distirbution) {
398  /*
399  Calculates the epsilon, the radius of the l2_ball in sopt
400  y:: vector for the l2 ball
401  */
402  if (distirbution == "chi^2") {
403  if (sigma == 0) {
404  return std::sqrt(2 * y_size + n_sigma * std::sqrt(4 * y_size));
405  }
406  return std::sqrt(2 * y_size + n_sigma * std::sqrt(4 * y_size)) * sigma;
407  }
408  if (distirbution == "chi") {
409  auto alpha =
410  1. / (8 * y_size) - 1. / (128 * y_size * y_size); // series expansion for gamma relation
411  auto mean = std::sqrt(2) * std::sqrt(y_size) *
412  (1 - alpha); // using Gamma(k+1/2)/Gamma(k) asymptotic formula
413  auto standard_deviation = std::sqrt(2 * y_size * alpha * (2 - alpha));
414  if (sigma == 0) {
415  return mean + n_sigma * standard_deviation;
416  }
417  return (mean + n_sigma * standard_deviation) * sigma;
418  }
419 
420  return 1.;

471 Vector<t_real> fit_fwhm(const Image<t_real> &psf, const t_int &size) {
472  /*
473  Find FWHM of point spread function, using least squares.
474 
475  psf:: point spread function, assumed to be normalised.
476 
477  The method assumes that you have sampled at least
478  3 pixels across the beam.
479  */
480 
481  auto x0 = std::floor(psf.cols() * 0.5);
482  auto y0 = std::floor(psf.rows() * 0.5);
483 
484  // finding patch
485  Image<t_real> patch =
486  psf.block(std::floor(y0 - size * 0.5) + 1, std::floor(x0 - size * 0.5) + 1, size, size);
487  PURIFY_LOW_LOG("Fitting to Patch");
488 
489  // finding values for least squares
490 
491  std::vector<t_tripletList> entries;
492  auto total_entries = 0;
493 
494  for (t_int i = 0; i < patch.cols(); ++i) {
495  for (t_int j = 0; j < patch.rows(); ++j) {
496  entries.emplace_back(j, i, std::log(patch(j, i)));
497  total_entries++;
498  }
499  }
500  Matrix<t_real> A = Matrix<t_real>::Zero(total_entries, 4);
501  Vector<t_real> q = Vector<t_real>::Zero(total_entries);
502  // putting values into a vector and matrix for least squares
503  for (t_int i = 0; i < total_entries; ++i) {
504  t_real x = entries.at(i).col() - size * 0.5 + 0.5;
505  t_real y = entries.at(i).row() - size * 0.5 + 0.5;
506  A(i, 0) = static_cast<t_real>(x * x); // x^2
507  A(i, 1) = static_cast<t_real>(x * y); // x * y
508  A(i, 2) = static_cast<t_real>(y * y); // y^2
509  q(i) = std::real(entries.at(i).value());
510  }
511  // least squares fitting
512  const auto solution =
513  static_cast<Vector<t_real>>(A.jacobiSvd(Eigen::ComputeThinU | Eigen::ComputeThinV).solve(q));
514  t_real const a = -solution(0);
515  t_real const b = +solution(1) * 0.5;
516  t_real const c = -solution(2);
517  // parameters of Gaussian
518  t_real theta = std::atan2(b, std::sqrt(std::pow(2 * b, 2) + std::pow(c - a, 2)) +
519  (c - a) * 0.5); // some relatively complicated trig identity to
520  // go from tan(2theta) to tan(theta).
521  t_real t = 0.;
522  t_real s = 0.;
523  if (std::abs(b) < 1e-13) {
524  t = a;
525  s = c;
526  theta = 0;
527  } else {
528  t = (a + c - 2 * b / std::sin(2 * theta)) * 0.5;
529  s = (a + c + 2 * b / std::sin(2 * theta)) * 0.5;
530  }
531 
532  t_real const sigma_x = std::sqrt(1 / (2 * t));
533  t_real const sigma_y = std::sqrt(1 / (2 * s));
534 
535  Vector<t_real> fit_parameters = Vector<t_real>::Zero(3);
536 
537  // fit for the beam rms, used for FWHM
538  fit_parameters(0) = sigma_x;

569 Array<t_complex> init_weights(const Vector<t_real> &u, const Vector<t_real> &v,
570  const Vector<t_complex> &weights, const t_real &oversample_factor,
571  const std::string &weighting_type, const t_real &R,
572  const t_int &ftsizeu, const t_int &ftsizev) {
573  /*
574  Calculate the weights to be applied to the visibilities in the measurement operator.
575  It does none, whiten, natural, uniform, and robust.
576  */
577  if (weighting_type == "none") return weights.array() * 0 + 1.;
578  if (weighting_type == "natural" or weighting_type == "whiten") return weights;
579  Vector<t_complex> out_weights(weights.size());
580  if ((weighting_type == "uniform") or (weighting_type == "robust")) {
581  t_real scale =
582  1. / oversample_factor; // scale for fov, controlling the region of sidelobe supression
583  Matrix<t_complex> gridded_weights = Matrix<t_complex>::Zero(ftsizev, ftsizeu);
584  for (t_int i = 0; i < weights.size(); ++i) {
585  t_int q = utilities::mod(floor(u(i) * scale), ftsizeu);
586  t_int p = utilities::mod(floor(v(i) * scale), ftsizev);
587  gridded_weights(p, q) += 1 * 1; // I get better results assuming all the weights are the
588  // same. It looks like miriad does this as well.
589  }
590  t_complex const sum_weights = (weights.array() * weights.array()).sum();
591  t_complex const sum_grid_weights2 = (gridded_weights.array() * gridded_weights.array()).sum();
592  t_complex const robust_scale =
593  sum_weights / sum_grid_weights2 * 25. *
594  std::pow(10, -2 * R); // Following standard formula, a bit different from miriad.
595  gridded_weights = gridded_weights.array().sqrt();
596  for (t_int i = 0; i < weights.size(); ++i) {
597  t_int q = utilities::mod(floor(u(i) * scale), ftsizeu);
598  t_int p = utilities::mod(floor(v(i) * scale), ftsizev);
599  if (weighting_type == "uniform") out_weights(i) = weights(i) / gridded_weights(p, q);
600  if (weighting_type == "robust") {
601  out_weights(i) = weights(i) / std::sqrt(1. + robust_scale * gridded_weights(p, q) *
602  gridded_weights(p, q));
603  }
604  }
605  } else
606  throw std::runtime_error("Wrong weighting type, " + weighting_type + " not recognised.");
607  return out_weights;
608 }

634 Matrix<t_complex> re_sample_ft_grid(const Matrix<t_complex> &input, const t_real &re_sample_ratio) {
635  /*
636  up samples image using by zero padding the fft
637 
638  input:: fft to be upsampled, with zero frequency at (0,0) of the matrix.
639 
640  */
641  if (re_sample_ratio == 1) return input;
642 
643  // sets up dimensions for old image
644  t_int old_x_centre_floor = std::floor(input.cols() * 0.5);
645  t_int old_y_centre_floor = std::floor(input.rows() * 0.5);
646  // need ceilling in case image is of odd dimension
647  t_int old_x_centre_ceil = std::ceil(input.cols() * 0.5);
648  t_int old_y_centre_ceil = std::ceil(input.rows() * 0.5);
649 
650  // sets up dimensions for new image
651  t_int new_x = std::floor(input.cols() * re_sample_ratio);
652  t_int new_y = std::floor(input.rows() * re_sample_ratio);
653 
654  t_int new_x_centre_floor = std::floor(new_x * 0.5);
655  t_int new_y_centre_floor = std::floor(new_y * 0.5);
656  // need ceilling in case image is of odd dimension
657  t_int new_x_centre_ceil = std::ceil(new_x * 0.5);
658  t_int new_y_centre_ceil = std::ceil(new_y * 0.5);
659 
660  Matrix<t_complex> output = Matrix<t_complex>::Zero(new_y, new_x);
661 
662  t_int box_dim_x;
663  t_int box_dim_y;
664 
665  // now have to move each quadrant into new grid
666  box_dim_x = std::min(old_x_centre_ceil, new_x_centre_ceil);
667  box_dim_y = std::min(old_y_centre_ceil, new_y_centre_ceil);
668  //(0, 0)
669  output.bottomLeftCorner(box_dim_y, box_dim_x) = input.bottomLeftCorner(box_dim_y, box_dim_x);
670 
671  box_dim_x = std::min(old_x_centre_ceil, new_x_centre_ceil);
672  box_dim_y = std::min(old_y_centre_floor, new_y_centre_floor);
673  //(y0, 0)
674  output.topLeftCorner(box_dim_y, box_dim_x) = input.topLeftCorner(box_dim_y, box_dim_x);
675 
676  box_dim_x = std::min(old_x_centre_floor, new_x_centre_floor);
677  box_dim_y = std::min(old_y_centre_ceil, new_y_centre_ceil);
678  //(0, x0)
679  output.bottomRightCorner(box_dim_y, box_dim_x) = input.bottomRightCorner(box_dim_y, box_dim_x);
680 
681  box_dim_x = std::min(old_x_centre_floor, new_x_centre_floor);
682  box_dim_y = std::min(old_y_centre_floor, new_y_centre_floor);
683  //(y0, x0)
684  output.topRightCorner(box_dim_y, box_dim_x) = input.topRightCorner(box_dim_y, box_dim_x);
685 
686  return output;
687 }
688 Matrix<t_complex> re_sample_image(const Matrix<t_complex> &input, const t_real &re_sample_ratio) {
689  const auto ft_plan = operators::fftw_plan::estimate;
690  auto fft =
691  purify::operators::init_FFT_2d<Vector<t_complex>>(input.rows(), input.cols(), 1., ft_plan);
692  Vector<t_complex> ft_grid = Vector<t_complex>::Zero(input.size());
693  std::get<0>(fft)(ft_grid, Vector<t_complex>::Map(input.data(), input.size()));
694  Matrix<t_complex> const new_ft_grid = utilities::re_sample_ft_grid(
695  Matrix<t_complex>::Map(ft_grid.data(), input.rows(), input.cols()), re_sample_ratio);
696  Vector<t_complex> output = Vector<t_complex>::Zero(input.size());
697  fft = purify::operators::init_FFT_2d<Vector<t_complex>>(new_ft_grid.rows(), new_ft_grid.cols(),
698  1., ft_plan);
699  std::get<1>(fft)(output, Vector<t_complex>::Map(new_ft_grid.data(), new_ft_grid.size()));
700  return Matrix<t_complex>::Map(output.data(), new_ft_grid.rows(), new_ft_grid.cols()) *
701  re_sample_ratio;
702 }
